# Richtlinie für den Zugriff von LLMs auf Reifencheck.de
## ------------------------------------------------------------
## Dieses Dokument definiert die Zugriffsrechte für KI-Crawler
## (Large Language Models) im Hinblick auf das Indexieren,
## Zusammenfassen und Verwenden der Inhalte von Reifencheck.de.
## Basierend auf Standards von llmcompliance.org.
## ------------------------------------------------------------

# -----------------------
# ZUGRIFFSREGELN FÜR KI-BOTS
# -----------------------

User-agent: GPTBot
Disallow: /                      ## Blockiert Trainingsdatenerfassung
Allow: /                         ## Erlaubt Indexierung für KI-Suche

User-agent: ClaudeBot
Disallow: /                      ## Blockiert Training von Claude-Modellen
Allow: /                         ## Erlaubt Retrieval/Suchindexierung

User-agent: PerplexityBot
Allow: /                         ## Erlaubt Suche, Antworten, Zitate

User-agent: CCBot
Disallow: /                      ## Blockiert Common-Crawl-Datensammlung

User-agent: AmazonBot
Disallow: /                      ## Blockiert Amazon Trainingsdaten

User-agent: YouBot
Allow: /                         ## Erlaubt You.com KI-Suche

User-agent: Google-Extended
Allow: /                         ## Erlaubt Indexierung für Google Gemini / SGE

User-agent: *
Allow: /                         ## Standard-Crawler erlaubt

# -----------------------
# KI-NUTZUNGSRICHTLINIEN
# -----------------------
Training: disallow               ## ❌ Keine Nutzung zur KI-Modell-Trainierung
Crawling: allow                  ## ✅ Indexierung für Suche & Zusammenfassungen erlaubt
Kommerziell: allow               ## ✅ Kommerzielle Nutzung in KI-Suchmaschinen erlaubt
Derivate: allow                  ## ✅ Zusammenfassungen, Antworten, Transformationen erlaubt
Attribution: required            ## ✅ Quellenangabe "Reifencheck.de" erforderlich
Personenbezogene-Daten: disallow ## ❌ Keine Speicherung/Verarbeitung personenbezogener Daten
Rate-Limit: reasonable           ## ⚠️ Verantwortliches Crawling erforderlich

Kontakt: support@reifencheck.de
Last-Modified: 2025-10-13
